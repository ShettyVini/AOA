#include<stdio.h>
int main()
{
	int a[5];
	int key;
	int i;
	int mid;
	int high=4;
	int low=0;
	printf("Enter elements:");
	for(i=1;i<=5;i++)
{
	scanf("%d",&a[i]);
}
printf("Enter key value:\n");
scanf("%d",&key);
while(low<=high)
{
	mid=(low+high)/2;
	if(a[mid]==key)
	{
		printf("Key found at %d",mid);
		return 0;
	}
	else if(a[mid]<key)
	{
		low=mid+1;
	}
	else{
		high=mid-1;
	}
}
return 0;
}
